from flask import Flask, render_template, request, send_file, after_this_request import os from pdf2docx import Converter app = Flask(__name__) # Usamos /tmp porque Render permite escribir archivos ahí sin errores UPLOAD_FOLDER = '/tmp' os.makedirs(UPLOAD_FOLDER, exist_ok=True) @app.route('/') def index(): return render_template('index.html') # Ruta corregida para coincidir con tu HTML original @app.route('/converter', methods=['POST']) def convertir(): archivo = request.files.get('archivo') if not archivo: return "No hay archivo", 400 ruta_entrada = os.path.join(UPLOAD_FOLDER, archivo.filename) archivo.save(ruta_entrada) nombre_base, extension = os.path.splitext(archivo.filename) if extension.lower() == ".pdf": ruta_salida = os.path.join(UPLOAD_FOLDER, nombre_base + ".docx") try: # Proceso de conversión cv = Converter(ruta_entrada) cv.convert(ruta_salida) cv.close() # Borra los archivos del servidor después de enviarlos para no llenar el disco @after_this_request def cleanup(response): try: if os.path.exists(ruta_entrada): os.remove(ruta_entrada) if os.path.exists(ruta_salida): os.remove(ruta_salida) except Exception as e: print(f"Error al limpiar: {e}") return response return send_file(ruta_salida, as_attachment=True) except Exception as e: return f"Error en la conversión: {str(e)}", 500 else: return "Solo se admiten archivos PDF", 400 if __name__ == '__main__': port = int(os.environ.get('PORT', 5000)) app.run(host='0.0.0.0', port=port) padding: 18px; border-radius: 15px; text-transform: uppercase; border: none; transition: 0.3s; } .btn-massive:hover { background-color: #e0e0e0; transform: translateY(-3px); }
